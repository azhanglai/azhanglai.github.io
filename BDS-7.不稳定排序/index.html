BDS-7.不稳定排序

发布于 2021-03-19 | 分类于 数据结构与算法-C版基础篇 | 4分钟 | 645字数 | 浏览量：

不稳定排序

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define SWAP(a, b) {\
    __typeof(a) _t = a;\
    a = b; b = _t;\
}

#define COLOR(a, b) "\e[" #b "m" a "\e[0m"
#define GREEN(a) COLOR(a, 32)

#define TEST(arr, n, func, args...) {\
    int *num = (int *)malloc(sizeof(int) * n);\
    memcpy(num, arr, sizeof(int) * n);\
    output(num, n);\
    printf("%s: ", #func);\
    func(args);\
    output(num, n);\
    printf("\n");\
    free(num);\
}
// 选择排序
/* 时间复杂度（最优，最坏，平均）O(n^2)
前：已排序，后：待排序
不稳定：例如 22'1 第一次后 12'2
每轮找一个最小的元素，与待排序区的第一个元素交换
*/
void select_sort(int *nums, int n) {
    for (int i = 0; i < n - 1; ++i) {
        int ind = i;
        for (int j = i; j < n; ++j) {
            if (nums[j] < nums[ind]) ind = j;
        }
        SWAP(nums[i], nums[ind]);
    }
    return ;
}
// 基础快排
/*时间复杂度：O(NlogN)
拿第一个元素作为基准值
逆序数列选第一个元素为基准值时，退化为选择排序，O（n^2）
*/
void quick_sort1(int *nums, int l, int r) {
    if (l >= r) return ;
    int x = l, y = r, z = nums[x];
    while (x < y) {
        while (x < y && nums[y] > z) y--;
        if (x < y) nums[x++] = nums[y];
        while (x < y && nums[x] < z) x++;
        if (x < y) nums[y--] = nums[x];
    }
    nums[x] = z;
    quick_sort1(nums, l, x - 1);
    quick_sort1(nums, x + 1, r);
    return ;
}
// 优化后的快排
void quick_sort2(int *nums, int l, int r) {
    // 优化1，减少一半递归，增加循环
    // 左半边用循环， 右半边用递归
    while (l < r) { 
        // 优化2 基准值选中间值
        int x = l, y = r, z = nums[(l + (r - l) / 2)];
        do {
            // 优化3：两个指针都去找值，并交换
            // 关键：x <= y,要用 <= 后面将 x,y错开
            // num[] < z,要用< , 经过基准值要停下
            while (x <= y && nums[x] < z) x++;
            while (x <= y && z < nums[y]) y--;
            if (x <= y) {
                // 交换后，两个指针都要移动一下，交错，y到前面，x到后面
                SWAP(nums[x], nums[y]);
                x++, y--; 
            }
        } while (x <= y); // 此处 用小于或小于等于都可以
        quick_sort2(nums, x, r); // 右半部分递归
        r = y; // l ~ y 左半部分循环
    }
    return ;
}

void randArr(int *nums, int n) {
    while (n--) nums[n] = rand() % 100;
    return ;
}

void output(int *nums, int n) {
    printf("arr: [");
    for (int i = 0; i < n; ++i) {
        i && printf(", ");
        printf("%d", nums[i]);
    }
    printf("]\n");
    return ;
}

int main() {
    srand(time(0));
    #define maxn 100
    clock_t start, end, end2;
    int arr[maxn];
    start = clock();
    TEST(arr, maxn, select_sort, num, maxn);
    randArr(arr, maxn);
    TEST(arr, maxn, quick_sort1, num, 0, maxn - 1);
    end = clock();
    printf(GREEN("time : %lf\n"), (double)(end - start) / ((clock_t)(1000)));
    TEST(arr, maxn, quick_sort2, num, 0, maxn - 1);
    end2 = clock();
    printf(GREEN("time : %lf\n"), (double)(end2 - end) / ((clock_t)(1000)));
    return 0;
}
