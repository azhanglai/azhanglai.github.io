C语言-2.数学运算

发布于 2020-10-25 | 分类于 C语言学习笔记 | 4分钟 | 774字数 | 浏览量：

1.C语言基本运算符

a = b; // = ： 赋值运算符
a = (b + c) * d; // +、-、*、/、（）：基本四则运算符
a = b % 2; // % ：求余运算符（取模运算符）
a = -b | c;  // &、|、^、~ ：位运算（二进制）
a = b >> 2; // <<、>> ：左移和右移（二进制）

2.C语言中的数学函数库

#include <math.h>  // 头文件
int  main() {
    int a, n;
    scanf("%d %d", &a, &n);
    int ans = pow(a, n); // 指数函数。
    //pow()函数原型：double pow(double a, double b);
    int ans = sqrt(n); // 开平方函数。
    //sqrt()函数原型：double sqrt(double x);
    int ans = fabs(n); // 实数绝对值函数。
    //fabs()函数原型：double fabs(double x);
    int ans = abs(n); // 头文件（stdlib.h），整数绝对值函数。
    //abs()函数原型：int abs(int x);
    int ans = ceil(n); // 上取整函数；
    //ceil()函数原型:double ceil(double x);
    int ans = floor(n); //下取整函数；
    //floor()函数原型：double floor(double x);
    int ans = log(n); // 以e为底对数函数。
    //log()函数原型：double log(double x);
    int ans = log10(n); // 以10为底对数函数。
    //log10()函数原型：double log10(double x);
    int ans = acos(n); // 反余弦函数(arccos())；
    //acos()函数原型：double acos(double x);
    //pi = acos(-1) = 3.1415923...
    return 0
}

3.练习题

1.请写一个程序，输入一个数字x, 输出数字x的立方根。

#include <stdio.h>
#include <math.h>

int main() {
    double x;
    scanf("%lf", &x);
    printf("%.2lf", pow(x, (1.0 / 3)));
    return 0;
}

2.请写一个程序，读入一个角度值，将角度值转为弧度值。

#include <stdio.h>
#include <math.h>
#define PI acos(-1)

int main() {
    double x;
    scanf("%lf", &x);
    printf("%.2lf", PI / 180.0 * x);
    return 0;
}

4.数学运算演示代码

#include <stdio.h>

int main() {
    int a = 7, b = 3, c;
    int *p = &a; // p是a的地址，*p是a的值。
    (*p)--; // (*p)-- = 7-1，a == 6。 *p--,右结合，值不变， a == 7。
    printf("a = %d\n", a);
    printf("a = %d, b = %d\n", a, b);
    //使用按位异或运算（^）进行值交换， swap
    a ^= b; // a = a ^ b
    b ^= a; // b = b ^ a ^ b = a
    a ^= b; // a = a ^ b ^ a = b
    printf("swap: a = %d, b = %d\n", a, b);
    return 0;
}
/*关于负数的二进制表示方法
0 ：0000 0000 0000 0000（16 位）
-1 == 0 - 1 ；1111 1111 1111 1111
1 ：0000 0000 0000 0001
1 的反码 ： 1111 1111 1111 1110
-1 == 1的反码加1
*/

5.inttypes头文件演示代码

#include <stdio.h>
#include <inttypes.h> //整数数据类型头文件，代码可移植性强。
//因为不同的系统，int所占的字节数是不同的，16位系统占2个字节，32和64位系统占4个字节
int main() {
    int32_t a = 70000;
    printf("%s\n", PRId32); // PRIdx是一个宏，x可以是（8，16，32，64）
    //在64位系统下，PRId32为 "d"。
    //printf("helloworld") == printf("hello" "wolrd");
    printf("%" PRId32 "\n", a); // printf("%d\n", a)
    printf("INT32_MIN = " "%" PRId32 "INT32_MAX = " "%" PRId32 "\n", INT32_MIN, INT32_MAX);
    //INTx_MIN：x可以是（8，16，32，64），最小值。
    //INT16_MIN：-32768，INT32_MIN：-2147483648
    //INTx_MAX：x可以是（8，16，32，63），最大值。
    //INT16_MAX：32767，INT32_MAX：2147483647
    return 0;
}
/*
有符号16位最大值二进制表示（第一位符号位）：0111 1111 1111 1111
最小值表示：1000 0000 0000 0000
*/
